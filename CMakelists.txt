CMAKE_MINIMUM_REQUIRED(VERSION 3.6)
PROJECT(PolarTracer LANGUAGES CUDA CXX)
ENABLE_LANGUAGE(CUDA)
SET(CMAKE_CUDA_STANDARD 17)
SET(CMAKE_CUDA_STANDARD_REQUIRED ON)
SET(CMAKE_CXX_STANDARD 17)
SET(CMAKE_CXX_STANDARD_REQUIRED ON)
SET(CMAKE_CUDA_ARCHITECTURES )
FILE(GLOB_RECURSE POLAR_TRACER_FILES "${CMAKE_SOURCE_DIR}/src/*.cuh" "${CMAKE_SOURCE_DIR}/src/*.cu")
ADD_EXECUTABLE(PolarTracer "${POLAR_TRACER_FILES}")
SET_PROPERTY(TARGET PolarTracer PROPERTY CUDA_ARCHITECTURES OFF)
SET_PROPERTY(DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR} PROPERTY VS_STARTUP_PROJECT PolarTracer)
